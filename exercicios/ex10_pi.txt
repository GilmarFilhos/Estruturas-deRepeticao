programa
{
    funcao inicio()
    {
        inteiro n, i, sinal
        real pi
        pi = 0
        sinal = 1

        escreva("Digite N termos: ")
        leia(n)

        para (i = 1; i <= n; i++)
        {
            pi = pi + (sinal * 1.0) / (2 * i - 1)
            sinal = sinal * -1
        }
        pi = pi * 4
        escreva("π ≈ ", pi, "\n")
    }
}
